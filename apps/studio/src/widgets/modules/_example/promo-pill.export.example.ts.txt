import type { WidgetNode } from '../../../domain/document/types';
import { escapeHtml, getBaseWidgetStyle } from '../../registry/export-helpers';
import type { ExportRendererManifestEntry } from '../export-registry';

export const promoPillExportRendererExample: ExportRendererManifestEntry = {
  type: 'x-promo-pill',
  render: ({ node }) => renderPromoPillExport(node as WidgetNode),
};

export function renderPromoPillExport(node: WidgetNode): string {
  const base = getBaseWidgetStyle(node);
  const title = escapeHtml(node.props.title ?? node.name);
  const caption = escapeHtml(node.props.caption ?? '');

  return `<div class="widget widget-promo-pill" data-widget-id="${node.id}" style="${base};flex-direction:column;gap:4px;">
    <strong>${title}</strong>
    ${caption ? `<span style="font-size:12px;opacity:.8;">${caption}</span>` : ''}
  </div>`;
}
